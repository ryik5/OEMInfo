создал ветку в реестре
HKEY_LOCAL_MACHINE\SOFTWARE\Policies\Microsoft\Windows NT\CurrentVersion\EFS
надо в efs записать параметр dword

rkHKLM = rkHKLM.OpenSubKey("Software");
rkHKLM = rkHKLM.OpenSubKey("Policies");
rkHKLM = rkHKLM.OpenSubKey("Microsoft");
rkHKLM = rkHKLM.OpenSubKey("Windows NT");
rkHKLM = rkHKLM.OpenSubKey("CurrentVersion");
rkHKLM = rkHKLM.OpenSubKey("EFS");    
rkHKLM.SetValue("EfsOptions",0,RegistryValueKind.DWord);
